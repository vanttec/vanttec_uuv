cmake_minimum_required(VERSION 3.0.2)
project(vanttec_uuv)

## Compile as C++11, supported in ROS Kinetic and newer
add_compile_options(-std=c++17)

find_package(catkin REQUIRED COMPONENTS
    roscpp
    message_generation
    std_msgs
    geometry_msgs
    nav_msgs
    tf2
    tf2_ros
    tf2_geometry_msgs
    vanttec_msgs # Not really shown on the directories
    visualization_msgs
)

add_message_files(FILES
   ThrustControl.msg
   GuidanceWaypoints.msg
   MasterStatus.msg
   Obstacle.msg
)

generate_messages(
  DEPENDENCIES
  std_msgs
  geometry_msgs
)

catkin_package(
  CATKIN_DEPENDS roscpp std_msgs geometry_msgs
  DEPENDS ${LIBS}
)

# Include contains the .hpp files
include_directories(
    uuv_gnc/include
    uuv_localization/include
    uuv_mission/include
    uuv_perception/include
    uuv_simulation/include
    ${catkin_INCLUDE_DIRS}
)

add_executable(uuv_odometry_node 
    uuv_odometry_node.cpp 
    odometry_calculator.cpp)
target_link_libraries(uuv_odometry_node ${catkin_LIBRARIES})
